VFIXUPIMMPD — 修正特殊打包 float64 值

操作码/指令 Op/En 64/32位模式支持 CPUID 特征标志 描述
EVEX.128.66.0F3A.W1 54 /r ib VFIXUPIMMPD xmm1 {k1}{z}, xmm2, xmm3/m128/m64bcst, imm8 A V/V AVX512VL AVX512F 在写掩码控制下，修正 float64 向量 xmm1、float64 向量 xmm2 和 int64 向量 xmm3/m128/m64bcst 中的特殊数值，并将结果存储在 xmm1 中。
EVEX.256.66.0F3A.W1 54 /r ib VFIXUPIMMPD ymm1 {k1}{z}, ymm2, ymm3/m256/m64bcst, imm8 A V/V AVX512VL AVX512F 在写掩码控制下，修正 float64 向量 ymm1、float64 向量 ymm2 和 int64 向量 ymm3/m256/m64bcst 中的特殊数值，并将结果存储在 ymm1 中。
EVEX.512.66.0F3A.W1 54 /r ib VFIXUPIMMPD zmm1 {k1}{z}, zmm2, zmm3/m512/m64bcst{sae}, imm8 A V/V AVX512F 使用 zmm3/m512/m64bcst 中的 int64 向量表修正 zmm2 中 float64 向量的元素，与 zmm1 中保留的元素合并，并将结果存储在 zmm1 中。

描述¶

使用第二个源操作数（第三个操作数）的对应四字元素中指定的 32 位二级查找表，对第一个源操作数（第二个操作数）中以双精度浮点格式编码的四字元素执行修正，并带有异常报告指定符 imm8。被修正的元素由程序掩码 k1 中指定的为 1 的掩码位选择。程序掩码 k1 中为 0 的掩码位或表响应操作为 0000b 会保留第一个操作数的对应元素。来自第一个源操作数的被修正元素与第一个操作数中被保留的元素合并，作为目的操作数（第一个操作数）的最终结果。

目的操作数和第一个源操作数是 ZMM/YMM/XMM 寄存器。第二个源操作数可以是一个 ZMM/YMM/XMM 寄存器、一个 512/256/128 位内存位置，或从一个 64 位内存位置广播的 512/256/128 位向量。

二级查找表通过将输入数据编码解码为 8 种标记（Token）类型，对第一个源操作数中的每个双精度浮点输入数据执行修正。为每种标记类型定义了一个响应表，用于将第一个源操作数中的输入编码转换为 16 种响应操作之一。

此指令专门用于修正涉及单个源的算术计算结果，使其符合规范，但它通常对于修正多指令序列的结果以反映特殊数字输入也很有用。例如，考虑 rcp(0)。根据 DX10 规范，向 rcp 输入 0 应得到 INF。然而，通过 Newton-Raphson 评估 rcp（其中 x=approx(1/0)）会产生错误的结果。为了解决这个问题，可以在 N-R 倒数序列之后使用 VFIXUPIMMPD 将结果设置为正确的值（即当输入为 0 时为 INF）。

如果 MXCSR.DAZ 动态屏蔽（DAZ）未设置，第一个源操作数中的非正规输入元素将被视为正常输入，不会触发任何修正或错误报告。

Imm8 用于设置所需的标志报告。它支持 #ZE 和 #IE 错误报告（详见下文）。

MXCSR 掩码位被忽略，并被视为所有掩码位都设置为屏蔽响应。如果设置了任何 imm8 位且满足错误报告条件，则可能会更新 MXCSR.IE 或 MXCSR.ZE。

此指令受写掩码保护，因此只有在向量掩码寄存器 k1 中设置了对应位的元素才会被计算并存储到 zmm1 中。目的操作数中对应位在 k1 中清零的元素将保留其先前的值或被设置为 0。

操作¶

enum TOKEN_TYPE
{
    QNAN_TOKEN := 0,
    SNAN_TOKEN := 1,
    ZERO_VALUE_TOKEN := 2,
    POS_ONE_VALUE_TOKEN := 3,
    NEG_INF_TOKEN := 4,
    POS_INF_TOKEN := 5,
    NEG_VALUE_TOKEN := 6,
    POS_VALUE_TOKEN := 7
}
FIXUPIMM_DP (dest[63:0], src1[63:0],tbl3[63:0], imm8 [7:0]){
    tsrc[63:0] := ((src1[62:52] = 0) AND (MXCSR.DAZ =1)) ? 0.0 : src1[63:0]
    CASE(tsrc[63:0] of TOKEN_TYPE) {
        QNAN_TOKEN: j := 0;
        SNAN_TOKEN: j := 1;
        ZERO_VALUE_TOKEN: j := 2;
        POS_ONE_VALUE_TOKEN: j := 3;
        NEG_INF_TOKEN: j := 4;
        POS_INF_TOKEN: j := 5;
        NEG_VALUE_TOKEN: j := 6;
        POS_VALUE_TOKEN: j := 7;
    } ; end source special CASE(tsrc...)
    ; The required response from src3 table is extracted
    token_response[3:0] = tbl3[3+4*j:4*j];
    CASE(token_response[3:0]) {
        0000: dest[63:0] := dest[63:0];
                ; preserve content of DEST
        0001: dest[63:0] := tsrc[63:0];
                ; pass through src1 normal input value, denormal as zero
        0010: dest[63:0] := QNaN(tsrc[63:0]);
        0011: dest[63:0] := QNAN_Indefinite;
        0100: dest[63:0] := -INF;
        0101: dest[63:0] := +INF;
        0110: dest[63:0] := tsrc.sign? –INF : +INF;
        0111: dest[63:0] := -0;
        1000: dest[63:0] := +0;
        1001: dest[63:0] := -1;
        1010: dest[63:0] := +1;
        1011: dest[63:0] := 1⁄2;
        1100: dest[63:0] := 90.0;
        1101: dest[63:0] := PI/2;
        1110: dest[63:0] := MAX_FLOAT;
        1111: dest[63:0] := -MAX_FLOAT;
    }
            ; end of token_response CASE
    ; The required fault reporting from imm8 is extracted
    ; TOKENs are mutually exclusive and TOKENs priority defines the order.
    ; Multiple faults related to a single token can occur simultaneously.
    IF (tsrc[63:0] of TOKEN_TYPE: ZERO_VALUE_TOKEN) AND imm8[0] then set #ZE;
    IF (tsrc[63:0] of TOKEN_TYPE: ZERO_VALUE_TOKEN) AND imm8[1] then set #IE;
    IF (tsrc[63:0] of TOKEN_TYPE: ONE_VALUE_TOKEN) AND imm8[2] then set #ZE;
    IF (tsrc[63:0] of TOKEN_TYPE: ONE_VALUE_TOKEN) AND imm8[3] then set #IE;
    IF (tsrc[63:0] of TOKEN_TYPE: SNAN_TOKEN) AND imm8[4] then set #IE;
    IF (tsrc[63:0] of TOKEN_TYPE: NEG_INF_TOKEN) AND imm8[5] then set #IE;
    IF (tsrc[63:0] of TOKEN_TYPE: NEG_VALUE_TOKEN) AND imm8[6] then set #IE;
    IF (tsrc[63:0] of TOKEN_TYPE: POS_INF_TOKEN) AND imm8[7] then set #IE;
        ; end fault reporting
    return dest[63:0];
}
        ; end of FIXUPIMM_DP()

VFIXUPIMMPD¶

(KL, VL) = (2, 128), (4, 256), (8, 512)
FOR j := 0 TO KL-1
    i := j * 64
    IF k1[j] OR *no writemask*
        THEN
            IF (EVEX.b = 1) AND (SRC2 *is memory*)
                THEN
                    DEST[i+63:i] := FIXUPIMM_DP(DEST[i+63:i], SRC1[i+63:i], SRC2[63:0], imm8 [7:0])
                ELSE
                    DEST[i+63:i] := FIXUPIMM_DP(DEST[i+63:i], SRC1[i+63:i], SRC2[i+63:i], imm8 [7:0])
            FI;
        ELSE
            IF *merging-masking* ; merging-masking
                THEN *DEST[i+63:i] remains unchanged*
                ELSE DEST[i+63:i] := 0
                        ; zeroing-masking
            FI
    FI;
ENDFOR
DEST[MAXVL-1:VL] := 0
