VFIXUPIMMSS — 修正特殊标量 float32 值

操作码/指令 Op/En 64/32位模式支持 CPUID 特征标志 描述
EVEX.LLIG.66.0F3A.W0 55 /r ib VFIXUPIMMSS xmm1 {k1}{z}, xmm2, xmm3/m32{sae}, imm8 A V/V AVX512F 使用 xmm3/m32 中的标量 int32 表修正 xmm2 的低双字元素中的 float32 数值，并将结果存储在 xmm1 中。

指令操作数编码¶

Op/En 元组类型 操作数 1 操作数 2 操作数 3 操作数 4
A Tuple1 Scalar ModRM:reg (r, w) EVEX.vvvv (r) ModRM:r/m (r) imm8

描述¶

使用第二个源操作数（第三个操作数）的低双字元素中指定的 32 位二级查找表，对第一个源操作数（第二个操作数）中以单精度浮点格式编码的低双字元素执行修正，并带有异常报告指定符 imm8。被修正的元素由程序掩码 k1 中指定的为 1 的掩码位选择。程序掩码 k1 中为 0 的掩码位或表响应操作为 0000b 会保留第一个操作数的对应元素。来自第一个源操作数的被修正元素或第一个操作数中被保留的元素成为目的操作数（第一个操作数）的低双字元素。目的操作数的 127:32 位从第一个源操作数的对应位复制。目的操作数和第一个源操作数是 XMM 寄存器。第二个源操作数可以是一个 XMM 寄存器或一个 32 位内存位置。

二级查找表通过将输入数据编码解码为 8 种标记（Token）类型，对第一个源操作数中的每个单精度浮点输入数据执行修正。为每种标记类型定义了一个响应表，用于将第一个源操作数中的输入编码转换为 16 种响应操作之一。

此指令专门用于修正涉及单个源的算术计算结果，使其符合规范，但它通常对于修正多指令序列的结果以反映特殊数字输入也很有用。例如，考虑 rcp(0)。根据 DX10 规范，向 rcp 输入 0 应得到 INF。然而，通过 Newton-Raphson 评估 rcp（其中 x=approx(1/0)）会产生错误的结果。为了解决这个问题，可以在 N-R 倒数序列之后使用 VFIXUPIMMPD 将结果设置为正确的值（即当输入为 0 时为 INF）。

如果 MXCSR.DAZ 动态屏蔽（DAZ）未设置，第一个源操作数中的非正规输入元素将被视为正常输入，不会触发 any 修正或错误报告。

Imm8 用于设置所需的标志报告。它支持 #ZE 和 #IE 错误报告（详见下文）。

使用 MXCSR.DAZ 且仅涉及 zmm2（即在设置了 MXCSR.DAZ 的情况下，zmm1 不被视为零）。

MXCSR 掩码位被忽略，并被视为所有掩码位都设置为屏蔽响应。如果设置了 any 的 imm8 位且满足错误报告条件，则可能会更新 MXCSR.IE 或 MXCSR.ZE。

操作¶

enum TOKEN_TYPE
{
    QNAN_TOKEN := 0,
    SNAN_TOKEN := 1,
    ZERO_VALUE_TOKEN := 2,
    POS_ONE_VALUE_TOKEN := 3,
    NEG_INF_TOKEN := 4,
    POS_INF_TOKEN := 5,
    NEG_VALUE_TOKEN := 6,
    POS_VALUE_TOKEN := 7
}
FIXUPIMM_SP (dest[31:0], src1[31:0],tbl3[31:0], imm8 [7:0]){
    tsrc[31:0] := ((src1[30:23] = 0) AND (MXCSR.DAZ =1)) ? 0.0 : src1[31:0]
    CASE(tsrc[63:0] of TOKEN_TYPE) {
        QNAN_TOKEN: j := 0;
        SNAN_TOKEN: j := 1;
        ZERO_VALUE_TOKEN: j := 2;
        POS_ONE_VALUE_TOKEN: j := 3;
        NEG_INF_TOKEN: j := 4;
        POS_INF_TOKEN: j := 5;
        NEG_VALUE_TOKEN: j := 6;
        POS_VALUE_TOKEN: j := 7;
    }
            ; end source special CASE(tsrc...)
    ; The required response from src3 table is extracted
    token_response[3:0] = tbl3[3+4*j:4*j];
    CASE(token_response[3:0]) {
        0000: dest[31:0] := dest[31:0];
        0001: dest[31:0] := tsrc[31:0];
        0010: dest[31:0] := QNaN(tsrc[31:0]);
        0011: dest[31:0] := QNAN_Indefinite;
        0100: dest[31:0] := -INF;
        0101: dest[31:0] := +INF;
        0110: dest[31:0] := tsrc.sign? –INF : +INF;
        0111: dest[31:0] := -0;
        1000: dest[31:0] := +0;
        1001: dest[31:0] := -1;
        1010: dest[31:0] := +1;
        1011: dest[31:0] := 1⁄2;
        1100: dest[31:0] := 90.0;
        1101: dest[31:0] := PI/2;
        1110: dest[31:0] := MAX_FLOAT;
        1111: dest[31:0] := -MAX_FLOAT;
    }
            ; end of token_response CASE
    ; The required fault reporting from imm8 is extracted
    ; TOKENs are mutually exclusive and TOKENs priority defines the order.
    ; Multiple faults related to a single token can occur simultaneously.
    IF (tsrc[31:0] of TOKEN_TYPE: ZERO_VALUE_TOKEN) AND imm8[0] then set #ZE;
    IF (tsrc[31:0] of TOKEN_TYPE: ZERO_VALUE_TOKEN) AND imm8[1] then set #IE;
    IF (tsrc[31:0] of TOKEN_TYPE: ONE_VALUE_TOKEN) AND imm8[2] then set #ZE;
    IF (tsrc[31:0] of TOKEN_TYPE: ONE_VALUE_TOKEN) AND imm8[3] then set #IE;
    IF (tsrc[31:0] of TOKEN_TYPE: SNAN_TOKEN) AND imm8[4] then set #IE;
    IF (tsrc[31:0] of TOKEN_TYPE: NEG_INF_TOKEN) AND imm8[5] then set #IE;
    IF (tsrc[31:0] of TOKEN_TYPE: NEG_VALUE_TOKEN) AND imm8[6] then set #IE;
    IF (tsrc[31:0] of TOKEN_TYPE: POS_INF_TOKEN) AND imm8[7] then set #IE;
        ; end fault reporting
    return dest[31:0];
} ; end of FIXUPIMM_SP()

    THEN DEST[31:0] := FIXUPIMM_SP(DEST[31:0], SRC1[31:0], SRC2[31:0], imm8 [7:0])
    ELSE
        IF *merging-masking* ; merging-masking
            THEN *DEST[31:0] remains unchanged*
            ELSE DEST[31:0] := 0
                ; zeroing-masking
        FI
FI;
