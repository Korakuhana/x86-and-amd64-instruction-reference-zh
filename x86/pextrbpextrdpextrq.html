描述¶

根据从 imm8[3:0] 确定的字节/双字/四字偏移量，从源 XMM 寄存器中提取一个字节/双字/四字整数值。目的操作数可以是寄存器或字节/双字/四字内存位置。如果目的操作数是寄存器，则寄存器的高位被零扩展。

在传统的非 VEX 编码版本中，如果目的操作数是寄存器，则 PEXTRB/PEXTRD 在 64 位模式下的默认操作数大小为 64 位，最低有效字节/双字数据之上的位被填充为零。PEXTRQ 在非 64 位模式下不可编码，且在 64 位模式下需要 REX.W。

注意：在 VEX.128 编码版本中，VEX.vvvv 是保留的且必须为 1111b，VEX.L 必须为 0，否则指令将导致 #UD。在 EVEX.128 编码版本中，EVEX.vvvv 是保留的且必须为 1111b，EVEX.L'L 必须为 0，否则指令将导致 #UD。如果目的操作数是寄存器，则 VPEXTRB/VPEXTRD 在 64 位模式下的默认操作数大小为 64 位，最低有效字节/字/双字数据之上的位被填充为零。

操作¶

CASE of
    PEXTRB: SEL := COUNT[3:0];
        TEMP := (Src >> SEL*8) AND FFH;
        IF (DEST = Mem8)
            THEN
            Mem8 := TEMP[7:0];
        ELSE IF (64-Bit Mode and 64-bit register selected)
            THEN
                R64[7:0] := TEMP[7:0];
                r64[63:8] := ZERO_FILL; };
        ELSE
                R32[7:0] := TEMP[7:0];
                r32[31:8] := ZERO_FILL; };
        FI;
    PEXTRD:SEL := COUNT[1:0];
        TEMP := (Src >> SEL*32) AND FFFF_FFFFH;
        DEST := TEMP;
    PEXTRQ: SEL := COUNT[0];
        TEMP := (Src >> SEL*64);
        DEST := TEMP;
EASC:

VPEXTRD/VPEXTRQ¶

IF (64-Bit Mode and 64-bit dest operand)
THEN
    Src_Offset := imm8[0]
    r64/m64 := (Src >> Src_Offset * 64)
ELSE
    Src_Offset := imm8[1:0]
    r32/m32 := ((Src >> Src_Offset *32) AND 0FFFFFFFFh);
FI

VPEXTRB（dest=m8）¶

SRC_Offset := imm8[3:0]
Mem8 := (Src >> Src_Offset*8)

VPEXTRB（dest=reg）¶

IF (64-Bit Mode )
THEN
    SRC_Offset := imm8[3:0]
    DEST[7:0] := ((Src >> Src_Offset*8) AND 0FFh)
    DEST[63:8] := ZERO_FILL;
ELSE
    SRC_Offset := imm8[3:0];
    DEST[7:0] := ((Src >> Src_Offset*8) AND 0FFh);
    DEST[31:8] := ZERO_FILL;
FI
